    <input type="number" id="averageClientValue">

    <input type="number" id="missedCallsPerMonth">

    <input type="number" id="averageCloseRate">

    <button onclick="calculateROI()">De Click Y Calcule Su Retorno De La Inversión</button>

    <h3>Results:</h3>

    <p>Cant. ($) mensuales Dejados En La Mesa: <span id="monthlyLeftOnTable"></span></p>

    <p>Nuestro Costo De Servicio: $127/mes <span id="amountCharged"></span></p>

    <p>Su Retorno De La Inversión: <span id="roi"></span></p>

  <script>

    function calculateROI() {

      const averageClientValue = parseFloat(document.getElementById("averageClientValue").value);

      const missedCallsPerMonth = parseFloat(document.getElementById("missedCallsPerMonth").value);

      const averageCloseRate = parseFloat(document.getElementById("averageCloseRate").value);

      const agencyFee = 127;


      const monthlyLeftOnTable = (missedCallsPerMonth * averageClientValue * (averageCloseRate / 100)) - agencyFee;

      const roi = ((monthlyLeftOnTable - agencyFee) / agencyFee) * 100;


      document.getElementById("monthlyLeftOnTable").innerHTML = `<span class="result">$${monthlyLeftOnTable.toFixed(2)}</span>`;

      document.getElementById("amountCharged").innerHTML = `<span class="result">$${agencyFee.toFixed(2)}</span>`;

      document.getElementById("roi").innerHTML = `<span class="result">${roi.toFixed(2)}%</span>`;

    }

  </script>
